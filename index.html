async function submitWaitlist(e, source) {

  e.preventDefault();

  const form = e.target;

  const input = form.querySelector('input[type="email"]');

  const submit = form.querySelector('button[type="submit"]');

  const email = input.value.trim();

 

  if(!email) return;

  submit.disabled = true;

  submit.textContent = 'Joining...';

  try {

    const response = await fetch(SCRIPT_URL, {

      method: 'POST',

      headers: {'Content-Type': 'text/plain'},

      body: JSON.stringify({

        type: 'waitlist',

        email: email,

        source: source

      })

    });

    if(source === 'main'){

      form.style.display = 'none';

      document.getElementById('successMsg').classList.add('show');

    } else {

      input.value = '';

      const note = document.getElementById('finalNote');

      note.innerHTML = '<strong style="color:var(--sage-light);">✓ You\'re on the list!</strong> Check your email.';

      submit.textContent = 'Join →';

      submit.disabled = false;

    }

  } catch(error) {

    console.error('Error:', error);

    alert('Oops! Something went wrong. Please try again.');

    submit.disabled = false;

    submit.textContent = source === 'main' ? 'Join Waitlist →' : 'Join →';

  }

}

async function submitFeature(e) {

  e.preventDefault();

  const form = e.target;

  const emailInput = document.getElementById('featureEmail');

  const textInput = document.getElementById('featureText');

  const submit = document.getElementById('featureSubmit');

 

  const email = emailInput.value.trim();

  const request = textInput.value.trim();

 

  if(!email || !request) return;

  submit.disabled = true;

  submit.textContent = 'Submitting...';

  try {

    const response = await fetch(SCRIPT_URL, {

      method: 'POST',

      headers: {'Content-Type': 'text/plain'},

      body: JSON.stringify({

        type: 'feature',

        email: email,

        request: request

      })

    });

    form.style.display = 'none';

    document.getElementById('featureSuccess').classList.add('show');

  } catch(error) {

    console.error('Error:', error);

    alert('Oops! Something went wrong. Please try again.');

    submit.disabled = false;

    submit.textContent = 'Submit Feature Request →';

  }

}
